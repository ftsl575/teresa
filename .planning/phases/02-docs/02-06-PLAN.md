    # Helper: resilient grep that tries the pathspec form first, falls back to client-side filter.
    # Single-quoted pathspec ':!.planning/' works in modern git but PowerShell escaping can mangle it.
    function Invoke-GitGrepExclPlanning {
        param([string]$Pattern, [string]$Mode = '-n', [string[]]$ExtraPathspecs = @())
        # Primary: native git pathspec (planning + any extras)
        $primaryArgs = @($Mode, $Pattern, '--', ':!.planning/') + $ExtraPathspecs
        $primary = & git grep @primaryArgs 2>$null
        if ($LASTEXITCODE -le 1) { return $primary }
        # Fallback: client-side filter (handles edge cases in PowerShell quoting)
        $all = & git grep $Mode $Pattern 2>$null
        $filtered = $all | Where-Object { $_ -notmatch '^\.planning/' }
        foreach ($extra in $ExtraPathspecs) {
            if ($extra -match '^:!(.+)$') {
                $exclude = $Matches[1]
                $filtered = $filtered | Where-Object { $_ -notmatch [regex]::Escape($exclude) }
            }
        }
        return $filtered
    }
